og:title */ -}} {{- $titleMatch := findRE `]+property=["']og:title["'][^>]+content=["']([^"']+)["']` $html 1 -}} {{- if not $titleMatch -}} {{- $titleMatch = findRE `]+content=["']([^"']+)["'][^>]+property=["']og:title["']` $html 1 -}} {{- end -}} {{- with $titleMatch -}} {{- $title = index . 0 | replaceRE `.*content=["']([^"']+)["'].*` "$1" -}} {{- end -}} {{- /* Fallback to */ -}} {{- if not $title -}} {{- $titleTagMatch := findRE `<title>([^<]+)` $html 1 -}} {{- with $titleTagMatch -}} {{- $title = index . 0 | replaceRE `([^<]+)` "$1" -}} {{- end -}} {{- end -}} {{- /* Extract og:description */ -}} {{- $descMatch := findRE `]+property=["']og:description["'][^>]+content=["']([^"']+)["']` $html 1 -}} {{- if not $descMatch -}} {{- $descMatch = findRE `]+content=["']([^"']+)["'][^>]+property=["']og:description["']` $html 1 -}} {{- end -}} {{- with $descMatch -}} {{- $description = index . 0 | replaceRE `.*content=["']([^"']+)["'].*` "$1" -}} {{- end -}} {{- /* Extract og:image */ -}} {{- $imageMatch := findRE `]+property=["']og:image["'][^>]+content=["']([^"']+)["']` $html 1 -}} {{- if not $imageMatch -}} {{- $imageMatch = findRE `]+content=["']([^"']+)["'][^>]+property=["']og:image["']` $html 1 -}} {{- end -}} {{- with $imageMatch -}} {{- $image = index . 0 | replaceRE `.*content=["']([^"']+)["'].*` "$1" -}} {{- end -}} {{- /* Extract og:site_name */ -}} {{- $siteMatch := findRE `]+property=["']og:site_name["'][^>]+content=["']([^"']+)["']` $html 1 -}} {{- if not $siteMatch -}} {{- $siteMatch = findRE `]+content=["']([^"']+)["'][^>]+property=["']og:site_name["']` $html 1 -}} {{- end -}} {{- with $siteMatch -}} {{- $siteName = index . 0 | replaceRE `.*content=["']([^"']+)["'].*` "$1" -}} {{- end -}} {{- end -}} {{- /* Parse domain from URL */ -}} {{- $domain := $url | replaceRE `^https?://([^/]+).*` "$1" -}} {{- /* Fallback title to domain */ -}} {{- if not $title -}} {{- $title = $domain -}} {{- end -}} {{- if $image -}} {{- end -}} {{ $title | htmlUnescape }} {{- if $description -}} {{ $description | htmlUnescape | truncate 120 }} {{- end -}} {{ $domain }} {{- if $siteName -}} {{ $siteName
